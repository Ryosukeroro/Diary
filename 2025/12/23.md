#ifndef GRID_UTILS_HPP
#define GRID_UTILS_HPP

#include "nav_msgs/OccupancyGrid.h"
#include <tuple>
#include <cmath>

// インライン関数にするか
//
inline std::tuple<int, int>get_grid_map_coord(double wx, double wy, const nav_msgs::OccupancyGrid& map) {
    int mx = static_cast<int>(std::floor((wx - map.info.origin.position.x) / map.info.resolution));
    int my = static_cast<int>(std::floor((wy - map.info.origin.position.y) / map.info.resolution));
    return {mx, my};
}

inline bool is_inside_map(int mx, int my, const nav_msgs::OccupancyGrid& map) {
    return (mx >= 0 && mx < map.info.width && my >= 0 && my < map.info.height);
}

#endif
